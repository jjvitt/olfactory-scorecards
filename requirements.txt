# ── 1. Shopify — Sales by POS location ────────────────────────────────────────
def fetch_shopify_sales():
    """Pull net sales + gross sales by POS location via ShopifyQL."""
    print("Fetching Shopify sales...")
    url = f"https://{SHOPIFY_STORE}/api/2024-01/graphql.json"
    headers = {"X-Shopify-Access-Token": SHOPIFY_TOKEN, "Content-Type": "application/json"}

    query = """
    {
      shopifyqlQuery(query: "FROM sales SHOW net_sales, gross_sales, discounts, returns GROUP BY pos_location_name SINCE %s UNTIL %s") {
        ... on TableResponse {
          tableData {
            rowData
            columns { name }
          }
        }
      }
    }
    """ % (YEAR_START, PERIOD_END)

    r = requests.post(url, json={"query": query}, headers=headers)
    data = r.json()
    rows = data["data"]["shopifyqlQuery"]["tableData"]["rowData"]
    cols = [c["name"] for c in data["data"]["shopifyqlQuery"]["tableData"]["columns"]]

    sales = {}
    for row in rows:
        d = dict(zip(cols, row))
        name = d.get("pos_location_name", "").strip()
        if name in STORE_CODES:
            sales[STORE_CODES[name]] = {
                "gross_sales": float(d.get("gross_sales", 0)),
                "net_sales":   float(d.get("net_sales", 0)),
                "discounts":   float(d.get("discounts", 0)),
                "returns":     float(d.get("returns", 0)),
            }
    print(f"  Got sales for {len(sales)} stores")
    return sales

# ── 2. Google Drive — Rippling CSVs ───────────────────────────────────────────
def get_google_services():
    creds = service_account.Credentials.from_service_account_info(
        GOOGLE_CREDS,
        scopes=[
            "https://www.googleapis.com/auth/drive.readonly",
            "https://www.googleapis.com/auth/spreadsheets.readonly",
        ]
    )
    drive  = build("drive",  "v3", credentials=creds)
    sheets = build("sheets", "v4", credentials=creds)
    return drive, sheets


def read_csv_from_drive(drive, folder_id, filename_pattern):
    """Find and download the most recent CSV matching a filename pattern."""
    results = drive.files().list(
        q=f"'{folder_id}' in parents and name contains '{filename_pattern}' and mimeType='text/csv'",
        orderBy="modifiedTime desc",
        pageSize=1,
        fields="files(id, name)"
    ).execute()
    files = results.get("files", [])
    if not files:
        raise FileNotFoundError(f"No CSV matching '{filename_pattern}' in Drive folder")
    file_id = files[0]["id"]
    print(f"  Reading: {files[0]['name']}")
    req = drive.files().get_media(fileId=file_id)
    buf = io.BytesIO()
    MediaIoBaseDownload(buf, req).next_chunk()
    buf.seek(0)
    return list(csv.DictReader(io.StringIO(buf.read().decode("utf-8-sig"))))


LOC_MAP = {
    "BLK":"BLK","ATL":"ATL","MOT":"MOT","Philadelphia":"PHL",
    "UNION":"UNION","BOS":"BOS","GEO":"GEO","WBG":"WBG","MIA":"MIA","DAL":"DAL"
}

def fetch_rippling_labor(drive):
    print("Fetching Rippling hours worked from Google Drive...")
    rows = read_csv_from_drive(drive, RIPPLING_FOLDER, "cost_by_employee")
    labor = defaultdict(lambda: {"pay": 0.0, "hours": 0.0, "emps": set()})
    for row in rows:
        emp = row.get("Employee", "").strip()
        loc = row.get("Location", "").strip()
        if emp == "All" or loc in ("All", "N/A", "OFFICE", "Remote", "CORP") or loc not in LOC_MAP:
            continue
        try:
            pay   = float(row.get("Gross pay", 0))
            hours = float(row.get("Time entry payable time (hours)", 0))
        except:
            continue
        if pay == 0:
            continue
        code = LOC_MAP[loc]
        labor[code]["pay"]   += pay
        labor[code]["hours"] += hours
        labor[code]["emps"].add(emp)
    print(f"  Got labor for {len(labor)} stores")
    return labor


# ── 3. Google Sheets — HB Event Revenue ──────────────────────────────────────
def fetch_event_revenue(sheets):
    """Sum deposit (col A) + remaining balance (col C), exclude rows where col G contains 'offsite'."""
    print("Fetching HB event revenue from Google Sheets...")
    # Get all sheet names
    meta = sheets.spreadsheets().get(spreadsheetId=HB_SHEET_ID).execute()
    sheet_names = [s["properties"]["title"] for s in meta["sheets"]]

    # Months to include: Jan–current month of current year
    months_to_include = []
    current = date.today()
    for m in range(1, current.month + 1):
        months_to_include.append(date(current.year, m, 1).strftime("%b"))  # "Jan", "Feb" etc.

    event_rev = defaultdict(float)
    for sheet in sheet_names:
        # Parse store code and month from sheet name e.g. "BLK Jan" or "Bleecker Jan 2026"
        code = None
        for s_code in STORE_CODES.values():
            if s_code.upper() in sheet.upper():
                code = s_code
                break
        if not code:
            for s_name, s_code in STORE_CODES.items():
                if any(w in sheet for w in s_name.split()):
                    code = s_code
                    break
        if not code:
            continue

        # Check month
        month_match = any(m in sheet for m in months_to_include)
        if not month_match:
            continue

        rows = sheets.spreadsheets().values().get(
            spreadsheetId=HB_SHEET_ID,
            range=f"'{sheet}'!A:G"
        ).execute().get("values", [])

        for row in rows[1:]:  # skip header
            if len(row) < 1:
                continue
            # Skip if col G (index 6) contains "offsite" (case-insensitive)
            col_g = row[6].strip().lower() if len(row) > 6 else ""
            if "offsite" in col_g:
                continue
            try:
                deposit   = float(str(row[0]).replace("$","").replace(",","")) if row[0] else 0
                remaining = float(str(row[2]).replace("$","").replace(",","")) if len(row) > 2 and row[2] else 0
                event_rev[code] += deposit + remaining
            except:
                continue

    print(f"  Got event revenue for {len(event_rev)} stores")
    return dict(event_rev)


# ── 4. Calculate metrics & generate data ─────────────────────────────────────
def calculate_store_data(sales, anc, labor, event_rev):
    stores = []
    for code in STORE_CODES.values():
        s = sales.get(code, {})
        a = anc.get(code, {})
        l = labor.get(code, {})
        ev = event_rev.get(code, 0.0)

        days     = PHL_DAYS if code == "PHL" else PERIOD_DAYS
        ann      = 365 / days

        net      = s.get("net_sales", 0) + ev
        gross    = s.get("gross_sales", 0)
        disc     = s.get("discounts", 0)
        ret      = s.get("returns", 0)
        labor_c  = l.get("pay", 0)
        hours    = l.get("hours", 0)
        hcount   = len(l.get("emps", set()))
        anc_net  = a.get("anc_net", 0)
        anc_ord  = a.get("anc_orders", 0)
        anc_pct  = anc_net / net * 100 if net else 0
        lcp      = labor_c / net * 100 if net else 0
        rplc     = net / labor_c if labor_c else 0
        splh     = net / hours if hours else 0
        disc_pct = abs(disc / gross * 100) if gross else 0
        lease    = LEASE.get(code, {})
        ann_s    = net * ann
        splr     = ann_s / lease["rent"]  if lease else 0
        splseat  = ann_s / SEATS[code]    if code in SEATS else 0

        stores.append({
            "code": code, "days": days, "ann": ann,
            "gross_sales": gross, "net_sales": net, "shopify_net": s.get("net_sales", 0),
            "discounts": disc, "returns": ret, "event_rev": ev,
            "labor_cost": labor_c, "hours": hours, "headcount": hcount,
            "anc_net": anc_net, "anc_orders": anc_ord, "anc_pct": anc_pct,
            "lcp": lcp, "rplc": rplc, "splh": splh, "disc_pct": disc_pct,
            "splr": splr, "splseat": splseat,
        })
    return stores


def rank_and_score(stores):
    WEIGHTS = {"rplc":30,"turn":25,"google":15,"anc_pct":15,"splr":10,"splseat":5}

    def rank_metric(key, higher_is_better):
        vals = [(s["code"], TURNOVER[s["code"]]["rateAnnual"] if key == "turn" else
                             RATINGS[s["code"]]["google"]     if key == "google" else
                             s[key]) for s in stores]
        vals.sort(key=lambda x: x[1], reverse=higher_is_better)
        return {code: i+1 for i, (code, _) in enumerate(vals)}

    ranks = {
        "rplc":    rank_metric("rplc",    True),
        "turn":    rank_metric("turn",    False),
        "google":  rank_metric("google",  True),
        "anc_pct": rank_metric("anc_pct", True),
        "splr":    rank_metric("splr",    True),
        "splseat": rank_metric("splseat", True),
    }

    N = len(stores)
    for s in stores:
        c = s["code"]
        s["ranks"] = {k: ranks[k][c] for k in ranks}
        s["score"] = round(sum(
            w * (N + 1 - ranks[k][c]) / N
            for k, w in WEIGHTS.items()
        ))

    stores.sort(key=lambda x: x["score"], reverse=True)
    return stores, ranks


# ── 5. Inject data into HTML template ────────────────────────────────────────
def generate_html(stores, period_label):
    """Read the base HTML and replace the data constants."""
    with open("olfactory_scorecards_template.html", "r") as f:
        html = f.read()

    def js_stores():
        lines = []
        for s in stores:
            name = next(k for k, v in STORE_CODES.items() if v == s["code"])
            city = {
                "BLK":"New York, NY","MOT":"New York, NY","WBG":"Brooklyn, NY",
                "PHL":"Philadelphia, PA","ATL":"Atlanta, GA","BOS":"Boston, MA",
                "UNION":"Washington, DC","GEO":"Washington, DC","MIA":"Miami, FL","DAL":"Dallas, TX"
            }[s["code"]]
            lines.append(
                f'  {{ name: "{name}", code: "{s["code"]}", city: "{city}", '
                f'gross_sales: {round(s["gross_sales"],2)}, net_sales: {round(s["net_sales"],2)}, '
                f'discounts: {round(s["discounts"],2)}, returns: {round(s["returns"],2)}, '
                f'labor_cost: {round(s["labor_cost"],2)}, hours: {round(s["hours"],1)}, '
                f'headcount: {s["headcount"]}, event_rev: {round(s["event_rev"],2)} }},'
            )
        return "[\n" + "\n".join(lines) + "\n]"

    # Replace key data blocks using markers in the template
    html = re.sub(
        r'// AUTO:PERIOD_LABEL.*?// END:PERIOD_LABEL',
        f'// AUTO:PERIOD_LABEL\nconst PERIOD_LABEL = "{period_label}";\n// END:PERIOD_LABEL',
        html, flags=re.DOTALL
    )
    html = re.sub(
        r'// AUTO:STORES.*?// END:STORES',
        f'// AUTO:STORES\nconst stores = {js_stores()};\n// END:STORES',
        html, flags=re.DOTALL
    )

    with open("olfactory_scorecards.html", "w") as f:
        f.write(html)
    print("HTML generated ✓")


# ── 6. Deploy to Netlify ──────────────────────────────────────────────────────
def deploy_to_netlify():
    print("Deploying to Netlify...")
    with open("olfactory_scorecards.html", "rb") as f:
        content = f.read()

    import hashlib
    sha = hashlib.sha1(content).hexdigest()

    headers = {
        "Authorization": f"Bearer {NETLIFY_TOKEN}",
        "Content-Type": "application/octet-stream",
    }

    # Create deploy
    r = requests.post(
        f"https://api.netlify.com/api/v1/sites/{NETLIFY_SITE_ID}/deploys",
        headers={"Authorization": f"Bearer {NETLIFY_TOKEN}", "Content-Type": "application/json"},
        json={"files": {"index.html": sha}, "async": False}
    )
    deploy = r.json()
    deploy_id = deploy["id"]
    print(f"  Deploy created: {deploy_id}")

    # Upload file
    r2 = requests.put(
        f"https://api.netlify.com/api/v1/deploys/{deploy_id}/files/index.html",
        headers=headers,
        data=content
    )
    print(f"  Upload status: {r2.status_code}")
    print(f"  Live at: https://{deploy.get('ssl_url', NETLIFY_SITE_ID + '.netlify.app')}")
